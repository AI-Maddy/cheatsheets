    from graphviz import Digraph
    from dataclasses import dataclass, field
    from typing import List, Optional
    from enum import Enum
    
    class GSNNodeType(Enum):
        GOAL = "goal"
        STRATEGY = "strategy"
        SOLUTION = "solution"
        CONTEXT = "context"
        ASSUMPTION = "assumption"
        JUSTIFICATION = "justification"
    
    @dataclass
    class GSNNode:
        id: str
        type: GSNNodeType
        text: str
        children: List['GSNNode'] = field(default_factory=list)
        contexts: List['GSNNode'] = field(default_factory=list)
        assumptions: List['GSNNode'] = field(default_factory=list)
        justifications: List['GSNNode'] = field(default_factory=list)
        
        def add_child(self, child: 'GSNNode'):
            self.children.append(child)
        
        def add_context(self, context: 'GSNNode'):
            self.contexts.append(context)
        
        def add_assumption(self, assumption: 'GSNNode'):
            self.assumptions.append(assumption)
        
        def add_justification(self, justification: 'GSNNode'):
            self.justifications.append(justification)
    
    class GSNDiagram:
        def __init__(self, title: str):
            self.title = title
            self.root: Optional[GSNNode] = None
            self.graph = None
        
        def set_root(self, root: GSNNode):
            self.root = root
        
        def _get_shape_and_style(self, node_type: GSNNodeType):
            """Return Graphviz shape and style for GSN node type"""
            styles = {
                GSNNodeType.GOAL: ("box", "filled", "lightblue"),
                GSNNodeType.STRATEGY: ("parallelogram", "filled", "lightyellow"),
                GSNNodeType.SOLUTION: ("ellipse", "filled", "lightgreen"),
                GSNNodeType.CONTEXT: ("box", "rounded,filled", "lightgray"),
                GSNNodeType.ASSUMPTION: ("box", "rounded,filled", "lightsalmon"),
                GSNNodeType.JUSTIFICATION: ("ellipse", "filled", "wheat")
            }
            return styles.get(node_type, ("box", "", "white"))
        
        def _add_node_to_graph(self, node: GSNNode):
            """Add a node to Graphviz graph"""
            shape, style, color = self._get_shape_and_style(node.type)
            
            # Wrap text for readability
            wrapped_text = self._wrap_text(f"{node.id}:\n{node.text}", 30)
            
            self.graph.node(
                node.id,
                wrapped_text,
                shape=shape,
                style=style,
                fillcolor=color,
                fontname="Arial",
                fontsize="10"
            )
        
        def _wrap_text(self, text: str, width: int) -> str:
            """Wrap text for node labels"""
            words = text.split()
            lines = []
            current_line = []
            current_length = 0
            
            for word in words:
                if current_length + len(word) + 1 <= width:
                    current_line.append(word)
                    current_length += len(word) + 1
                else:
                    if current_line:
                        lines.append(' '.join(current_line))
                    current_line = [word]
                    current_length = len(word)
            
            if current_line:
                lines.append(' '.join(current_line))
            
            return '\n'.join(lines)
        
        def _build_graph(self, node: GSNNode):
            """Recursively build Graphviz graph"""
            # Add main node
            self._add_node_to_graph(node)
            
            # Add contexts (dashed lines)
            for ctx in node.contexts:
                self._add_node_to_graph(ctx)
                self.graph.edge(ctx.id, node.id, style="dashed", arrowhead="none")
            
            # Add assumptions (dashed lines)
            for assump in node.assumptions:
                self._add_node_to_graph(assump)
                self.graph.edge(assump.id, node.id, style="dashed", arrowhead="none")
            
            # Add justifications (dashed lines)
            for just in node.justifications:
                self._add_node_to_graph(just)
                self.graph.edge(just.id, node.id, style="dashed", arrowhead="none")
            
            # Add children (solid lines)
            for child in node.children:
                self.graph.edge(node.id, child.id, arrowhead="vee")
                self._build_graph(child)
        
        def render(self, filename='gsn_diagram', format='png'):
            """Render GSN diagram to file"""
            self.graph = Digraph(comment=self.title)
            self.graph.attr(rankdir='TB')  # Top to bottom
            self.graph.attr('node', fontname='Arial')
            
            if self.root:
                self._build_graph(self.root)
            
            self.graph.render(filename, format=format, cleanup=True)
            print(f"GSN diagram rendered to {filename}.{format}")

**Q1:** What are the six core GSN symbols and their purposes?

**A1:**

1. **Goal** (rectangle): Claim to be substantiated
2. **Strategy** (parallelogram/diamond): How to decompose goal
3. **Solution** (circle): Evidence supporting goal
4. **Context** (rounded box): Clarifying information

**Q2:** What is the difference between Context and Assumption in GSN?

**A2:**

- **Context**: Clarifying information that is TRUE (e.g., "ODD = urban roads")
  - Defines scope, environment, definitions
  - Not disputed

- **Assumption**: Statement ASSUMED to be true but not proven in this argument
  - Example: "Operator is trained"
  - Risk: If assumption is false, argument may break
  - Should be minimized and tracked
